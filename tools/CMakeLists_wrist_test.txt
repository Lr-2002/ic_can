# CMakeLists.txt for Wrist Kinematics Test
cmake_minimum_required(VERSION 3.10)
project(WristKinematicsTest)

# Set C++ standard
set(CMAKE_CXX_STANDARD 14)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Create executable for wrist kinematics test
add_executable(test_wrist_kinematics test_wrist_kinematics.cpp)

# Link math library if needed
if(UNIX AND NOT APPLE)
    target_link_libraries(test_wrist_kinematics m)
endif()

# Set output directory
set_target_properties(test_wrist_kinematics PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin
)
