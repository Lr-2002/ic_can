import pandas as pd
import matplotlib.pyplot as plt
df = pd.read_csv('validation_results.csv')
for joint in range(6):
    joint_data = df[df['joint'] == joint]
    plt.figure()
    plt.plot(joint_data['actual_torque'], label='Actual')
    plt.plot(joint_data['predicted_torque'], label='Predicted')
    plt.title(f'Joint {joint} Torque Validation')
    plt.legend()
    plt.show()
